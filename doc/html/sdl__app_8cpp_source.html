152void SDLApp::render_grid() {
153 window_->clear(0, 0, 0);
154
155 int win_w = window_->get_width();
156 int win_h = window_->get_height();
157 int cell_w = win_w / 2;
158 int cell_h = win_h / 2;
159
160 // Grid positions: [0]=top-left, [1]=top-right, [2]=bottom-left, [3]=bottom-right
161 int cell_x[4] = {0, cell_w, 0, cell_w};
162 int cell_y[4] = {0, 0, cell_h, cell_h};
163
164 // Find global min/max temperature
165 double global_min = 1e9, global_max = -1e9;
166
167 if (sim_type_ == SimType::BAR_1D) {
168 for (int i = 0; i < 4; i++) {
169 if (solvers_1d_[i]) {
170 auto temps = solvers_1d_[i]->get_temperature();
171 for (double t : temps) {
172 global_min = std::min(global_min, t);
173 global_max = std::max(global_max, t);
174 }
175 }
176 }
177 } else {
178 for (int i = 0; i < 4; i++) {
179 if (solvers_2d_[i]) {
180 auto temps = solvers_2d_[i]->get_temperature_2d();
181 for (const auto& row : temps) {
182 for (double t : row) {
183 global_min = std::min(global_min, t);
184 global_max = std::max(global_max, t);
185 }
186 }
187 }
188 }
189 }
190
191 // Add margin to range
192 double margin = (global_max - global_min) * 0.05;
193 heatmap_->set_range(global_min - margin, global_max + margin);
194
195 // Render each material in its cell
196 for (int i = 0; i < 4; i++) {
197 SimInfo info;
198 info.material_name = materials_[i].name;
199 info.alpha = materials_[i].alpha();
200 info.L = L_;
201 info.tmax = tmax_;
202 info.u0 = u0_ + 273.15;
203 info.speed = speed_;
204 info.paused = paused_;
205
206 if (sim_type_ == SimType::BAR_1D && solvers_1d_[i]) {
207 info.time = solvers_1d_[i]->get_time();
208 auto temps = solvers_1d_[i]->get_temperature();
209 if (!temps.empty()) {
210 heatmap_->draw_1d_cell(temps, info, cell_x[i], cell_y[i], cell_w, cell_h);
211 }
212 } else if (sim_type_ == SimType::PLATE_2D && solvers_2d_[i]) {
213 info.time = solvers_2d_[i]->get_time();
214 auto temps = solvers_2d_[i]->get_temperature_2d();
215 if (!temps.empty() && !temps[0].empty()) {
216 heatmap_->draw_2d_cell(temps, info, cell_x[i], cell_y[i], cell_w, cell_h);
217 }
218 }
219 }
220
221 // Draw grid separators
222 SDL_Renderer* rend = window_->get_renderer();
223 SDL_SetRenderDrawColor(rend, 255, 255, 255, 255);
224 // Vertical separator (3 pixels wide)
225 for (int dx = -1; dx <= 1; dx++) {
226 SDL_RenderDrawLine(rend, cell_w + dx, 0, cell_w + dx, win_h);
227 }
228 // Horizontal separator (3 pixels wide)
229 for (int dy = -1; dy <= 1; dy++) {
230 SDL_RenderDrawLine(rend, 0, cell_h + dy, win_w, cell_h + dy);
231 }
232
233 // Add corner markers
234 SDL_SetRenderDrawColor(rend, 200, 200, 200, 255);
235 int corner_size = 10;
236
237 // Center cross highlight
238 SDL_RenderDrawLine(rend, cell_w - corner_size, cell_h, cell_w + corner_size, cell_h);
239 SDL_RenderDrawLine(rend, cell_w, cell_h - corner_size, cell_w, cell_h + corner_size);
240
241 window_->present();
242}

244void SDLApp::process_events(SDL_Event& event) {
245 if (event.type == SDL_KEYDOWN) {
246 switch (event.key.keysym.sym) {
247 case SDLK_ESCAPE:
248 running_ = false;
249 break;
250 case SDLK_SPACE:
251 paused_ = !paused_;
252 break;
253 case SDLK_r:
254 if (grid_mode_) {
255 for (int i = 0; i < 4; i++) {
256 if (solvers_1d_[i]) solvers_1d_[i]->reset();
257 if (solvers_2d_[i]) solvers_2d_[i]->reset();
258 }
259 } else {
260 if (solver_1d_) solver_1d_->reset();
261 if (solver_2d_) solver_2d_->reset();
262 }
263 paused_ = false;
264 break;
265 case SDLK_UP:
266 speed_ = std::min(sim_type_ == SimType::BAR_1D ? 50 : 20, speed_ + 5);
267 break;
268 case SDLK_DOWN:
269 speed_ = std::max(1, speed_ - 5);
270 break;
271 }
272 }

275void SDLApp::run() {
276 while (running_) {
277 SDL_Event event;
278 while (SDL_PollEvent(&event)) {
279 if (event.type == SDL_QUIT ||
280 (event.type == SDL_WINDOWEVENT &&
281 event.window.event == SDL_WINDOWEVENT_CLOSE)) {
282 running_ = false;
283 }
284 process_events(event);
285 }
286
287 if (!running_) break;
288
289 if (!paused_) {
290 for (int s = 0; s < speed_; s++) {
291 if (grid_mode_) {
292 bool all_done = true;
293 for (int i = 0; i < 4; i++) {
294 if (sim_type_ == SimType::BAR_1D && solvers_1d_[i]) {
295 if (solvers_1d_[i]->step()) all_done = false;
296 } else if (sim_type_ == SimType::PLATE_2D && solvers_2d_[i]) {
297 if (solvers_2d_[i]->step()) all_done = false;
298 }
299 }
300 if (all_done) {
301 paused_ = true;
302 break;
303 }
304 } else {
305 if (sim_type_ == SimType::BAR_1D && solver_1d_) {
306 if (!solver_1d_->step()) {
307 paused_ = true;
308 break;
309 }
310 } else if (sim_type_ == SimType::PLATE_2D && solver_2d_) {
311 if (!solver_2d_->step()) {
312 paused_ = true;
313 break;
314 }
315 }
316 }
317 }
318 }
319
320 if (grid_mode_) {
321 render_grid();
322 } else {
323 render();
324 }
325 SDLCore::delay(16);
326 }
327
328 heatmap_.reset();
329 window_.reset();
330}
