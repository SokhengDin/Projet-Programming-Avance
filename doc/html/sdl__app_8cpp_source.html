78void SDLApp::process_events(SDL_Event& event) {
79 if (event.type == SDL_KEYDOWN) {
80 switch (event.key.keysym.sym) {
81 case SDLK_ESCAPE:
82 running_ = false;
83 break;
84 case SDLK_SPACE:
85 paused_ = !paused_;
86 break;
87 case SDLK_r:
88 if (solver_1d_) solver_1d_->reset();
89 if (solver_2d_) solver_2d_->reset();
90 paused_ = false;
91 break;
92 case SDLK_UP:
93 speed_ = std::min(sim_type_ == SimType::BAR_1D ? 50 : 20, speed_ + 5);
94 break;
95 case SDLK_DOWN:
96 speed_ = std::max(1, speed_ - 5);
97 break;
98 }
99 }

102void SDLApp::run() {
103 while (running_) {
104 SDL_Event event;
105 while (SDL_PollEvent(&event)) {
106 if (event.type == SDL_QUIT ||
107 (event.type == SDL_WINDOWEVENT &&
108 event.window.event == SDL_WINDOWEVENT_CLOSE)) {
109 running_ = false;
110 }
111 process_events(event);
112 }
113
114 if (!running_) break;
115
116 if (!paused_) {
117 for (int i = 0; i < speed_; i++) {
118 if (sim_type_ == SimType::BAR_1D && solver_1d_) {
119 if (!solver_1d_->step()) {
120 paused_ = true;
121 break;
122 }
123 } else if (sim_type_ == SimType::PLATE_2D && solver_2d_) {
124 if (!solver_2d_->step()) {
125 paused_ = true;
126 break;
127 }
128 }
129 }
130 }
131
132 render();
133 SDLCore::delay(16);
134 }
135
136 heatmap_.reset();
137 window_.reset();
138}
