20void SDLHeatmap::auto_range(const std::vector<double>& temps) {
21 if (temps.empty()) return;
22 auto minmax = std::minmax_element(temps.begin(), temps.end());
23 double margin = (*minmax.second - *minmax.first) * 0.05;
24 t_min_ = *minmax.first - margin;
25 t_max_ = *minmax.second + margin;
26 if (t_max_ - t_min_ < 1.0) {
27 t_min_ -= 0.5;
28 t_max_ += 0.5;
29 }
30}

32void SDLHeatmap::auto_range_2d(const std::vector<std::vector<double>>& temps) {
33 if (temps.empty()) return;
34 double min_v = temps[0][0];
35 double max_v = temps[0][0];
36 for (const auto& row : temps) {
37 for (double v : row) {
38 min_v = std::min(min_v, v);
39 max_v = std::max(max_v, v);
40 }
41 }
42 double margin = (max_v - min_v) * 0.05;
43 t_min_ = min_v - margin;
44 t_max_ = max_v + margin;
45 if (t_max_ - t_min_ < 1.0) {
46 t_min_ -= 0.5;
47 t_max_ += 0.5;
48 }
49}

88void SDLHeatmap::temp_to_rgb(double t, Uint8& r, Uint8& g, Uint8& b) const {
89 double norm = (t - t_min_) / (t_max_ - t_min_);
90 norm = std::max(0.0, std::min(1.0, norm));
91
92 double idx = norm * (INFERNO_SIZE - 1);
93 int i0 = static_cast<int>(idx);
94 int i1 = std::min(i0 + 1, INFERNO_SIZE - 1);
95 double frac = idx - i0;
96
97 r = static_cast<Uint8>(INFERNO_MAP[i0][0] * (1 - frac) + INFERNO_MAP[i1][0] * frac);
98 g = static_cast<Uint8>(INFERNO_MAP[i0][1] * (1 - frac) + INFERNO_MAP[i1][1] * frac);
99 b = static_cast<Uint8>(INFERNO_MAP[i0][2] * (1 - frac) + INFERNO_MAP[i1][2] * frac);
100}

102void SDLHeatmap::draw_1d_fullscreen(const std::vector<double>& temps) {
103 if (temps.empty()) return;
104
105 SDL_Renderer* rend = win_.get_renderer();
106 int win_w = win_.get_width();
107 int win_h = win_.get_height();
108 int n = static_cast<int>(temps.size());
109
110 for (int i = 0; i < n; i++) {
111 Uint8 r, g, b;
112 temp_to_rgb(temps[i], r, g, b);
113
114 int x1 = (i * win_w) / n;
115 int x2 = ((i + 1) * win_w) / n;
116
117 SDL_SetRenderDrawColor(rend, r, g, b, 255);
118 SDL_Rect rect = {x1, 0, x2 - x1 + 1, win_h};
119 SDL_RenderFillRect(rend, &rect);
120 }

123void SDLHeatmap::draw_2d_fullscreen(const std::vector<std::vector<double>>& temps) {
124 if (temps.empty()) return;
125
126 SDL_Renderer* rend = win_.get_renderer();
127 int win_w = win_.get_width();
128 int win_h = win_.get_height();
129
130 int ny = static_cast<int>(temps.size());
131 int nx = static_cast<int>(temps[0].size());
132
133 // Bilinear interpolation with subsampling
134 int sub = 2;
135 int render_nx = (nx - 1) * sub;
136 int render_ny = (ny - 1) * sub;
137
138 for (int sj = 0; sj < render_ny; ++sj) {
139 for (int si = 0; si < render_nx; ++si) {
140 double fi = static_cast<double>(si) / sub;
141 double fj = static_cast<double>(sj) / sub;
142
143 int i0 = static_cast<int>(fi);
144 int j0 = static_cast<int>(fj);
145 int i1 = std::min(i0 + 1, nx - 1);
146 int j1 = std::min(j0 + 1, ny - 1);
147
148 double fx = fi - i0;
149 double fy = fj - j0;
150
151 double t = temps[j0][i0] * (1-fx) * (1-fy)
152 + temps[j0][i1] * fx * (1-fy)
153 + temps[j1][i0] * (1-fx) * fy
154 + temps[j1][i1] * fx * fy;
155
156 Uint8 r, g, b;
157 temp_to_rgb(t, r, g, b);
158
159 int x1 = (si * win_w) / render_nx;
160 int x2 = ((si + 1) * win_w) / render_nx;
161 int y1 = (sj * win_h) / render_ny;
162 int y2 = ((sj + 1) * win_h) / render_ny;
163
164 SDL_SetRenderDrawColor(rend, r, g, b, 255);
165 SDL_Rect rect = {x1, y1, x2 - x1 + 1, y2 - y1 + 1};
166 SDL_RenderFillRect(rend, &rect);
167 }
168 }
