139int main() {
140 bool running = true;
141
142 while (running) {
143 print_header();
144
145 int sim_type = select_simulation_type();
146 if (sim_type == 0) {
147 std::cout << "\nExit.\n";
148 break;
149 }
150 if (sim_type < 1 || sim_type > 4) {
151 std::cout << "\nInvalid choice.\n";
152 continue;
153 }
154
155 double L = 1.0, tmax = 16.0, u0 = 13.0, f = 80.0;
156
157 // Grid mode (options 3 and 4)
158 if (sim_type == 3 || sim_type == 4) {
159 if (!get_parameters(L, tmax, u0, f)) continue;
160 if (!confirm_and_start_grid(sim_type, L, tmax, u0, f)) continue;
161
162 std::cout << "\nStarting grid simulation...\n";
163
164 try {
165 sdl::SDLCore::init();
166
167 sdl::SDLApp::SimType type = (sim_type == 3)
168 ? sdl::SDLApp::SimType::BAR_1D
169 : sdl::SDLApp::SimType::PLATE_2D;
170
171 sdl::SDLApp app(type, L, tmax, u0, f); // Grid mode constructor
172 app.run();
173
174 sdl::SDLCore::quit();
175 std::cout << "\nReturning to menu...\n";
176
177 } catch (const std::exception& e) {
178 std::cerr << "Error: " << e.what() << std::endl;
179 sdl::SDLCore::quit();
180 }
181 }
182 // Single material mode (options 1 and 2)
183 else {
184 int material_idx = select_material();
185 if (material_idx < 0) continue;
186
187 if (!get_parameters(L, tmax, u0, f)) continue;
188 if (!confirm_and_start(sim_type, material_idx, L, tmax, u0, f)) continue;
189
190 ensiie::Material material;
191 switch (material_idx) {
192 case 0: material = ensiie::Materials::COPPER; break;
193 case 1: material = ensiie::Materials::IRON; break;
194 case 2: material = ensiie::Materials::GLASS; break;
195 case 3: material = ensiie::Materials::POLYSTYRENE; break;
196 default: material = ensiie::Materials::COPPER; break;
197 }
198
199 std::cout << "\nStarting simulation...\n";
200
201 try {
202 sdl::SDLCore::init();
203
204 sdl::SDLApp::SimType type = (sim_type == 1)
205 ? sdl::SDLApp::SimType::BAR_1D
206 : sdl::SDLApp::SimType::PLATE_2D;
207
208 sdl::SDLApp app(type, material, L, tmax, u0, f);
209 app.run();

216 sdl::SDLCore::quit();
217 }
218 }
219 }
