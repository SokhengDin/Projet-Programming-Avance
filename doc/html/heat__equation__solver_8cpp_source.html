49/**
50 * @brief Setup heat sources F(x)
51 * F = tmax·f² on [L/10, 2L/10] and [5L/10, 6L/10]
52 */
53void HeatEquationSolver1D::init_source(double f) {
54 // Source regions: [L/10, 2L/10] and [5L/10, 6L/10]
55 // F(x) = tmax * f^2 according to PDF
56 double f1 = tmax_ * f * f;
57 double f2 = 0.75 * tmax_ * f * f;
58
59 // Scale factor to make heat propagation visible
60 double scale = 100.0; // Amplification factor for visualization
61
62 for (int i = 0; i < n_; i++) {
63 double x = i * dx_;
64 if (x >= L_ / 10.0 && x <= 2.0 * L_ / 10.0) {
65 F_[i] = f1 * scale;
66 } else if (x >= 5.0 * L_ / 10.0 && x <= 6.0 * L_ / 10.0) {
67 F_[i] = f2 * scale;
68 } else {
69 F_[i] = 0.0;
70 }
71 }
72}

74/**
75 * @brief Advance one time step using implicit scheme
76 * @return false if t >= tmax
77 */
78bool HeatEquationSolver1D::step() {
79 if (t_ >= tmax_) return false;
80
81 // Implicit scheme coefficients
82 double alpha = mat_.alpha();
83 double r = alpha * dt_ / (dx_ * dx_);
84 double coef = dt_ / (mat_.rho * mat_.c);
85
86 std::vector<double> a(n_, -r);
87 std::vector<double> b(n_, 1.0 + 2.0 * r);
88 std::vector<double> c(n_, -r);
89 std::vector<double> d(n_);
90
91 // RHS
92 for (int i = 0; i < n_; i++) {
93 d[i] = u_[i] + coef * F_[i];
94 }
95
96 // Neumann BC at x=0
97 b[0] = 1.0 + r;
98 c[0] = -r;
99
100 // Dirichlet BC at x=L
101 b[n_ - 1] = 1.0;
102 a[n_ - 1] = 0.0;
103 c[n_ - 1] = 0.0;
104 d[n_ - 1] = u0_kelvin_;
105
106 std::vector<double> u_new(n_);
107 solve_tridiagonal(a, b, c, d, u_new);
108
109 u_ = u_new;
110 t_ += dt_;
111 return true;
112}

114/**
115 * @brief Thomas algorithm (TDMA) for tridiagonal system
116 * Complexity: O(n)
117 */
118void HeatEquationSolver1D::solve_tridiagonal(
119 const std::vector<double>& a,
120 const std::vector<double>& b,
121 const std::vector<double>& c,
122 std::vector<double>& d,
123 std::vector<double>& x
124) {
125 // Thomas algorithm (TDMA)
126 int n = static_cast<int>(b.size());
127 std::vector<double> c_prime(n);
128 std::vector<double> d_prime(n);
129
130 // Forward elimination
131 c_prime[0] = c[0] / b[0];
132 d_prime[0] = d[0] / b[0];
133
134 for (int i = 1; i < n; i++) {
135 double denom = b[i] - a[i] * c_prime[i - 1];
136 c_prime[i] = c[i] / denom;
137 d_prime[i] = (d[i] - a[i] * d_prime[i - 1]) / denom;
138 }
139
140 // Back substitution
141 x[n - 1] = d_prime[n - 1];
142 for (int i = n - 2; i >= 0; --i) {
143 x[i] = d_prime[i] - c_prime[i] * x[i + 1];
144 }
145}

191/**
192 * @brief Setup 4 symmetric heat sources at corners
193 * F = tmax·f² on [L/6, 2L/6]² and [4L/6, 5L/6]² regions
194 */
195void HeatEquationSolver2D::init_source(double f) {
196 // Four symmetric sources at corners
197 // F(x,y) = tmax * f^2 according to PDF
198 double f_val = tmax_ * f * f;
199
200 // Scale factor to make heat propagation visible
201 double scale = 100.0; // Amplification factor for visualization
202
203 for (int j = 0; j < n_; j++) {
204 for (int i = 0; i < n_; i++) {
205 double x = i * dx_;
206 double y = j * dx_;
207
208 bool in_source = false;
209
210 // Bottom-left
211 if (x >= L_/6.0 && x <= 2.0*L_/6.0 && y >= L_/6.0 && y <= 2.0*L_/6.0)
212 in_source = true;
213 // Bottom-right
214 if (x >= 4.0*L_/6.0 && x <= 5.0*L_/6.0 && y >= L_/6.0 && y <= 2.0*L_/6.0)
215 in_source = true;
216 // Top-left
217 if (x >= L_/6.0 && x <= 2.0*L_/6.0 && y >= 4.0*L_/6.0 && y <= 5.0*L_/6.0)
218 in_source = true;
219 // Top-right
220 if (x >= 4.0*L_/6.0 && x <= 5.0*L_/6.0 && y >= 4.0*L_/6.0 && y <= 5.0*L_/6.0)
221 in_source = true;
222
223 F_[idx(i, j)] = in_source ? (f_val * scale) : 0.0;
224 }
225 }
226}

228/**
229 * @brief Advance one step using Gauss-Seidel iteration
230 * @return false if t >= tmax
231 */
232bool HeatEquationSolver2D::step() {
233 if (t_ >= tmax_) return false;
234
235 // Implicit scheme with 5-point stencil
236 double alpha = mat_.alpha();
237 double r = alpha * dt_ / (dx_ * dx_);
238 double src_coef = dt_ / (mat_.rho * mat_.c);
239
240 std::vector<double> u_new = u_;
241
242 // Gauss-Seidel parameters
243 const int max_iter = 100;
244 const double tol = 1e-6;
245
246 for (int iter = 0; iter < max_iter; iter++) {
247 double max_diff = 0.0;
248
249 for (int j = 0; j < n_; ++j) {
250 for (int i = 0; i < n_; ++i) {
251 // Dirichlet BC at right and top edges
252 if (i == n_ - 1 || j == n_ - 1) {
253 u_new[idx(i, j)] = u0_kelvin_;
254 continue;
255 }
256
257 double old_val = u_new[idx(i, j)];
258
259 // Neighbors with Neumann BC (mirror at i=0, j=0)
260 double u_left = (i > 0) ? u_new[idx(i-1, j)] : u_new[idx(1, j)];
261 double u_right = u_new[idx(i+1, j)];
262 double u_down = (j > 0) ? u_new[idx(i, j-1)] : u_new[idx(i, 1)];
263 double u_up = u_new[idx(i, j+1)];
264
265 double rhs = u_[idx(i, j)] + src_coef * F_[idx(i, j)];
266 u_new[idx(i, j)] = (rhs + r * (u_left + u_right + u_down + u_up)) / (1.0 + 4.0 * r);
267
268 max_diff = std::max(max_diff, std::abs(u_new[idx(i, j)] - old_val));
269 }
270 }
271
272 if (max_diff < tol) break;
273 }
274
275 u_ = u_new;
276 t_ += dt_;
277 return true;
278}
